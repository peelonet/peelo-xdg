49 template<class T = std::filesystem::path>
50 std::vector<T>
51 get_multiple_dirs(const char* env_variable_name)
52 {
53 std::vector<T> result;
54
55 if (auto begin = std::getenv(env_variable_name))
56 {
57 auto end = begin;
58
59 for (; *end; ++end)
60 {
61 if (*end != path_separator)
62 {
63 continue;
64 }
65 else if (end - begin > 0)
66 {
67 result.emplace_back(std::string(begin, end - begin));
68 }
69 begin = end + 1;
70 }
71 if (end - begin > 0)
72 {
73 result.emplace_back(std::string(begin, end - begin));
74 }
75 }
76
77 return result;
78 }
79 }

84 template<class T = std::filesystem::path>
85 std::optional<T>
86 home_dir()
87 {
88 const auto home = std::getenv("HOME");
89
90 if (!home)
91 {
92#if defined(_WIN32)
93 const auto userprofile = std::getenv("USERPROFILE");
94
95 if (!userprofile)
96 {
97 const auto homedrive = std::getenv("HOMEDRIVE");
98 const auto homepath = std::getenv("HOMEPATH");
99
100 if (homedrive && *homedrive && homepath && *homepath)
101 {
102 return std::filesystem::path(homedrive) / homepath;
103 }
104 } else {
105 return userprofile;
106 }
107#else
108 const auto pw = ::getpwuid(::getuid());
109
110 if (pw)
111 {
112 return pw->pw_dir;
113 }
114#endif
115
116 return std::nullopt;
117 }
118
119 return home;
120 }
